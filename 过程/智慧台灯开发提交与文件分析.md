**E53_SC1.c** — PWM 亮度调节与按键中断：
```c
+static uint8_t g_brightness = 0;
+static uint8_t g_lightStatus = 0;
+
+static int32_t ButtonIrqCallback(uint16_t gpio, void *data)
+{
+    if (g_lightStatus == 0) {
+        g_lightStatus = 1;
+        g_brightness = 80;
+        E53_SC1SetBrightness(g_brightness);
+    } else {
+        g_lightStatus = 0;
+        g_brightness = 0;
+        E53_SC1SetBrightness(0);
+    }
+    return 0;
+}
+
+int E53_SC1SetBrightness(uint8_t brightness)
+{
+    uint32_t duty;
+    if (brightness > E53_SC1_BRIGHTNESS_MAX) {
+        brightness = E53_SC1_BRIGHTNESS_MAX;
+    }
+    g_brightness = brightness;
+    duty = (uint32_t)((float)brightness / 100.0f * E53_SC1_PWM_MAX_DUTY);
+    E53_PWMSetDuty(duty);
+    if (brightness > 0) {
+        g_lightStatus = 1;
+    } else {
+        g_lightStatus = 0;
+    }
+    return 0;
+}
```

**根因**：初始 PWM period 设为 `1000000`（纳秒级），但 STM32MP1 PWM HDF 驱动将 period/duty 直接写入 TIM 计数器，`1000000` 超过 16 位 TIM 最大值，导致溢出。

**E53_SC1.c** — 按键轮询线程与状态机：
```c
+#include "gpio_if.h"
+#include "osal_thread.h"
+
+#define E53_SC1_BUTTON_THREAD_STACK 0x1000
+#define E53_SC1_BUTTON_SCAN_MS 20
+#define E53_SC1_BUTTON_DEBOUNCE_MS 300
+#define E53_SC1_BUTTON_DOUBLE_MS 400
+#define E53_SC1_BUTTON_LONG_MS 800
+#define E53_SC1_BUTTON_ADJUST_MS 180
+#define E53_SC1_BRIGHTNESS_STEP 5
+#define E53_SC1_BRIGHTNESS_DIM_MIN 0
+#define E53_SC1_DEFAULT_BRIGHTNESS 80
+
+typedef enum {
+    E53_SC1_KEY_IDLE = 0,
+    E53_SC1_KEY_FIRST_DOWN,
+    E53_SC1_KEY_WAIT_SECOND,
+    E53_SC1_KEY_SECOND_DOWN,
+    E53_SC1_KEY_LONG_BRIGHTEN,
+    E53_SC1_KEY_LONG_DIM,
+} E53_SC1KeyState;
+
+static void E53_SC1ButtonToggleLight(void)
+{
+    if (g_lightStatus) {
+        E53_SC1LightStatusSet(OFF);
+    } else {
+        E53_SC1LightStatusSet(ON);
+    }
+}
+
+static void E53_SC1ButtonAdjustBrightness(int direction)
+{
+    int brightness;
+    if (!g_lightStatus && direction > 0) {
+        E53_SC1LightStatusSet(ON);
+    }
+    if (g_brightness == 0) {
+        g_brightness = g_lastBrightness;
+    }
+    brightness = g_brightness + (direction * E53_SC1_BRIGHTNESS_STEP);
+    if (direction > 0 && brightness > E53_SC1_BRIGHTNESS_MAX) {
+        brightness = E53_SC1_BRIGHTNESS_MAX;
+    }
+    if (direction < 0 && brightness < E53_SC1_BRIGHTNESS_DIM_MIN) {
+        brightness = E53_SC1_BRIGHTNESS_DIM_MIN;
+    }
+    E53_SC1SetBrightness((uint8_t)brightness);
+}
+
+static int E53_SC1ButtonThread(void *arg)
+{
+    E53_SC1KeyState state = E53_SC1_KEY_IDLE;
+    uint32_t pressMs = 0, waitMs = 0, adjustMs = 0;
+
+    while (g_buttonThreadRunning) {
+        uint16_t level = E53_Level_High;
+        if (GpioRead(E53_SC1_BUTTON_GPIO, &level) == 0) {
+            uint8_t pressed = (level == E53_Level_Down) ? 1 : 0;
+            switch (state) {
+                case E53_SC1_KEY_IDLE:
+                    if (pressed) { pressMs = 0; state = E53_SC1_KEY_FIRST_DOWN; }
+                    break;
+                case E53_SC1_KEY_FIRST_DOWN:
+                    if (pressed) {
+                        pressMs += E53_SC1_BUTTON_SCAN_MS;
+                        if (pressMs >= E53_SC1_BUTTON_LONG_MS) {
+                            E53_SC1ButtonAdjustBrightness(1);  // 长按增亮
+                            state = E53_SC1_KEY_LONG_BRIGHTEN;
+                        }
+                    } else {
+                        waitMs = 0; state = E53_SC1_KEY_WAIT_SECOND;
+                    }
+                    break;
+                case E53_SC1_KEY_WAIT_SECOND:
+                    if (pressed) {
+                        pressMs = 0; state = E53_SC1_KEY_SECOND_DOWN;
+                    } else {
+                        waitMs += E53_SC1_BUTTON_SCAN_MS;
+                        if (waitMs >= E53_SC1_BUTTON_DOUBLE_MS) {
+                            E53_SC1ButtonToggleLight();  // 单击切换
+                            state = E53_SC1_KEY_IDLE;
+                        }
+                    }
+                    break;
+                case E53_SC1_KEY_SECOND_DOWN:
+                    if (pressed) {
+                        pressMs += E53_SC1_BUTTON_SCAN_MS;
+                        if (pressMs >= E53_SC1_BUTTON_LONG_MS) {
+                            E53_SC1ButtonAdjustBrightness(-1);  // 双击长按减暗
+                            state = E53_SC1_KEY_LONG_DIM;
+                        }
+                    } else { state = E53_SC1_KEY_IDLE; }
+                    break;
+                case E53_SC1_KEY_LONG_BRIGHTEN:
+                case E53_SC1_KEY_LONG_DIM:
+                    if (pressed) {
+                        adjustMs += E53_SC1_BUTTON_SCAN_MS;
+                        if (adjustMs >= E53_SC1_BUTTON_ADJUST_MS) {
+                            E53_SC1ButtonAdjustBrightness(state == E53_SC1_KEY_LONG_BRIGHTEN ? 1 : -1);
+                            adjustMs = 0;
+                        }
+                    } else { state = E53_SC1_KEY_IDLE; }
+                    break;
+            }
+        }
+        OsalMSleep(E53_SC1_BUTTON_SCAN_MS);
+    }
+    return 0;
+}
+
+int E53_SC1ButtonInit(void)
+{
+    GpioSetDir(E53_SC1_BUTTON_GPIO, GPIO_DIR_IN);
+    g_buttonThreadRunning = 1;
+    OsalThreadCreate(&g_buttonThread, E53_SC1ButtonThread, NULL);
+    OsalThreadStart(&g_buttonThread, &threadCfg);
+    return 0;
+}
```

**根因**：`Stm32Mp157GpioSetIrq` 的回调函数被忽略（`(void)func`），GPIO IRQ 不可靠，改用 OSAL 线程轮询。

**huawei_mqtt_demo.c** — MQTT 连接与上报：
```c
+#define MQTT_HOST "bb4de50857.iotda-device.cn-south-4.myhuaweicloud.com"
+#define MQTT_PORT 1883
+#define DEVICE_ID "6a2fab136b6c4d5f8d63cf9e_bearpi_sc1_001"
+#define MQTT_CLIENT_ID "6a2fab136b6c4d5f8d63cf9e_bearpi_sc1_001_0_0_2026061508"
+#define MQTT_USERNAME DEVICE_ID
+#define MQTT_PASSWORD ""   // 运行时传入
+#define SERVICE_ID "LightService"
+#define E53_SC1_SERVICE "hdf_e53_sc1"
+
+static int g_mqttPassword = MQTT_PASSWORD;
+
+// 主函数
+int main(int argc, char **argv) {
+    const char *host = MQTT_HOST;
+    if (argc > 1) host = argv[1];
+    if (argc > 2) g_mqttPassword = argv[2];  // 运行时传入密码
+
+    int fd = ConnectTcp(host, MQTT_PORT);
+    SendMqttConnect(fd);           // CONNECT + CONNACK
+    SendMqttSubscribe(fd);         // SUBSCRIBE properties/set/#
+    E53Sc1Init();                  // 初始化 HDF 硬件
+
+    while (1) {
+        select(fd, &readSet, NULL, NULL, &timeout);  // 等待下发
+        HandleMqttPacket(fd);      // 处理下发消息
+        E53Sc1Read(&lamp);         // 读取传感器
+        SendPropertyReport(fd, &lamp);  // 上报属性
+        SendPingReq(fd);           // 保活
+    }
+}
+
+// 下发处理
+static int HandlePropertySet(int fd, const char *topic, const char *payload) {
+    char requestId[128];
+    ExtractRequestId(topic, requestId, sizeof(requestId));
+
+    if (JsonIntValue(payload, "led_brightness", &brightness) == 0) {
+        E53Sc1SetBrightness(brightness);   // 设置亮度
+        E53Sc1SendCommand(E53_SC1_CMD_SET_LIGHT, brightness > 0 ? "ON" : "OFF", ...);
+    }
+    if (JsonBoolValue(payload, "led_status", &ledStatus) == 0) {
+        E53Sc1SetLight(ledStatus);         // 设置开关
+    }
+
+    SendSetResponse(fd, requestId, 0);     // 回复平台
+    return 0;
+}
+
+// 回复 set response
+static int SendSetResponse(int fd, const char *requestId, int resultCode) {
+    char topic[320];
+    snprintf(topic, sizeof(topic),
+        "$oc/devices/%s/sys/properties/set/response/request_id=%s", DEVICE_ID, requestId);
+    char payload[128];
+    snprintf(payload, sizeof(payload),
+        "{\"result_code\":%d,\"result_desc\":\"%s\"}", resultCode, "success");
+    return SendMqttPublishRaw(fd, topic, payload);
+}
+
+// 上报属性
+static int SendPropertyReport(int fd, const LampData *lamp) {
+    int reportLedOn = lamp->ledOn || lamp->brightness > 0;
+    snprintf(payload, sizeof(payload),
+        "{\"services\":[{\"service_id\":\"%s\",\"properties\":{\"light\":%.2f,\"led_status\":%s,\"led_brightness\":%d}}]}",
+        SERVICE_ID, lamp->lux, reportLedOn ? "true" : "false", lamp->brightness);
+    return SendMqttPublishRaw(fd, topic, payload);
+}
```
